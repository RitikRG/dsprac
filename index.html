# q8

class Queue:
  def __init__(self,data):
    self.data=data
  def display(self):
    print(self.data)
  def enqueue(self,value):
    self.data.append(value)
  def dequeue(self):
    n=self.data[0]
    self.data.pop(0)
    return n

q=Queue([1,2,3,4,5])
q.display()
q.enqueue(6)
q.display()
print(q.dequeue())
q.display()

# q12

def factorial(n):
  if n<=1:
    return 1
  return n*factorial(n-1)

print(factorial(5))

# q13

def fibonacci_tail(n,b,a):
  if n<=1:
    return b
  else:
    return fibonacci_tail(n-1,b+a,b)


def fibonacci(n):
  if n<=0:
    return -1
  elif n==1:
    return 0
  elif n==2:
    return 1
  return fibonacci_tail(n,1,0)
 
print(fibonacci(100))
